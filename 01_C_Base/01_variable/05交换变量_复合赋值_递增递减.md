//交换变量！
#include <stdio.h>


int main(){
	
	int a=5;
	int b=6;
	int t;//多定义一个变量，相当于临时存放的地方。 
	
	t=a;  
	a=b;
	b=t;
	printf("a=%d,b=%d\n",a,b);
	
	
	
	
	return 0;
}

{
