//循环控制
//素数	只能被1和自己整除的数，不包括!	 2,3,5,7,11,13,17, 19...

#include <stdio.h>

int main(){
	
	int x;
	
	scanf("%d", &x);
	
	int i;
	int isPrime =1;  //x是素数 
	
	for ( i= 2; i<x; i++){
			if( x % i == 0){
			isPrime =0;
			//break;	//break 跳出循环   
			continue; 	//continue 跳过循环这一轮剩下的语句进入下一轮. 
		}
		printf("%d\n",i);
	}
	if (isPrime ==1){
		printf("是素数\n");
	}else{
		printf("不是素数\n");
	}
	return 0;
}
