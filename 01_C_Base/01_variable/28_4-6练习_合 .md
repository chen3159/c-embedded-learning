
//约分 最简公式 
#include <stdio.h>
int main(){
	
	int dividend, divisor;
	scanf("%d/%d",&dividend, &divisor);
	
	int a = dividend;
	int b = divisor;
	int t;													//	a    b      t
	while ( b>0  ){		//辗转相除法求最大公约数, 			//	6    8    	6
		t = a % b;											//	8    6      2	
		a = b;												//	6    2      0
		b = t;												//	2	 0		
	} 
	
	printf("%d/%d\n",dividend/a,divisor/a);
		
	return 0;
}


/*5-2.念数字(15)
输入一个整数，输出每个数字对应的拼音。当整数为负数时，先输出“fu”字。十个数字对应的拼音如下:
0: ling
1: yi
2: er
3: san
4: si
5: wu
6: liu
7:qi
8: ba
9: jiu
输入格式:
输入在一行中给出一个整数，如:1234。
提示:整数包括负数、零和正数。
输出格式:*/

#include <stdio.h>
int main(){
	int x;
	scanf("%d",&x);
	
	if ( x<0 ){
		printf("fu ");		 // 先输出“负”的拼音
		x = -x;		// 把负数变成正数，方便后续处理每一位数字
	}
	
	int mask = 1;
	int t = x;		
	while ( t>9 ){		  // 循环：计算mask的值，让mask和数字的最高位对齐	// 例如数字123，循环结束后mask=100
		t /= 10;			// 去掉最后一位		 
		mask *= 10;
	}
	
	do {			// 取出当前最高位的数字
		int d = x / mask;
		
		switch ( d ){
			case 0:printf("ling"); break;
			case 0:printf("yi"); break;
			case 0:printf("er"); break;
			case 0:printf("san"); break;
			case 0:printf("si"); break;
			case 0:printf("wu"); break;
			case 0:printf("liu"); break;
			case 0:printf("qi"); break;
			case 0:printf("ba"); break;
			case 0:printf("ling"); break;
		}
		 // 如果不是最后一位数字，输出空格分隔
		if ( mask > 9 ) printf (" ");
		// 去掉已经处理过的最高位数字
		x %=mask;
		// 掩码缩小10倍，处理下一位
		mask /=10;
	}while ( mask >0 );
	printf("\n");
	 
	
	return 0;
}


//求a的连续和
 
#include <stdio.h>
int main(){
	
	int a,n;
	scanf("%d %d", &a, &n);
	
	int sum = 0;	 // sum用来存储最终的累加和，初始化为0	
	int i;
	int t =  0;		// t用来存储每一项的值（如2,22,222...），初始化为0
	
	for ( i=0; i<n; i++){
		t=  t*10 +a ;	// 生成当前项：每次把之前的数左移一位（乘10），再加上a
		sum += t;		// 把当前项t累加到总和sum中
	}
	printf("%d\n",sum);
	
	return 0;
}
