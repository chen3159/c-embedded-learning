这段代码的主要功能是使用一种增量筛选的方法，找出前 10 个质数（素数）并按格式输出。

![image-20260402211335349](https://raw.githubusercontent.com/czh010/images/main/hardwa/image-20260402211335349.png)

```c
#include <stdio.h>

// 函数声明：判断 x 是否为质数
// 参数：x 为待判断的数，knownPrimes 为已知质数数组，numberOfKnownPrimes 为已知质数的个数
int isPrime(int x, int knownPrimes[], int numberOfKnownPrimes);

int main(void)
{
    const int number = 10; // 目标：找出前 10 个质数
    int prime[number] = {2}; // 初始化质数数组，第一个元素为 2（第一个质数）
    int count = 1; // 当前已找到的质数个数（初始为 1，因为已包含 2）
    int i = 3; // 从 3 开始检查后续的数

    // 循环直到找到 10 个质数
    while (count < number) {
        // 调用 isPrime 判断 i 是否为质数
        if (isPrime(i, prime, count)) {
            prime[count++] = i; // 如果是质数，存入数组并增加计数
        }

        // 【调试代码】输出当前检查的数 i 和已找到的质数个数 count
        printf("i=%d \tcnt=%d\t", i, count);
        int i; // 这里重新定义了 i（会覆盖外层的 i，可能导致逻辑错误）
        // 遍历并输出当前质数数组的所有元素
        for (i = 0; i < number; i++) {
            printf("%d\t", prime[i]);
        }
        printf("\n");
    }

    i++; // 检查下一个数（注意：这里的 i++ 在 while 循环外，实际应放在循环内）

    // 最终输出质数数组（每 5 个换行）
    for (i = 0; i < number; i++) {
        printf("%d", prime[i]);
        if ((i + 1) % 5) {
            printf("\t"); // 每 5 个质数后换行，否则用制表符分隔
        } else {
            printf("\n");
        }
    }
    return 0;
}

// isPrime 函数定义（代码中未给出，需补充）
// 功能：判断 x 是否为质数，通过检查是否能被已知质数整除
int isPrime(int x, int knownPrimes[], int numberOfKnownPrimes) {
    for (int i = 0; i < numberOfKnownPrimes; i++) {
        if (x % knownPrimes[i] == 0) {
            return 0; // 能被已知质数整除，不是质数
        }
    }
    return 1; // 不能被任何已知质数整除，是质数
}
```
