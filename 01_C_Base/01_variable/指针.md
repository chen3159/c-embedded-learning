```c
#include <stdio.h>

// 函数声明
void f(int *p); // 接收一个整型指针（地址）
void g(int k);  // 接收一个整型值（副本）

int main(void) {
    int i = 6; // 在内存中开辟空间存储 6

    // 打印变量 i 的内存地址，例如 0xbffbcd70
    printf("&i=%p\n", &i);

    // 【传址调用】：把 i 的地址传递给函数 f
    // 此时 f 内部的 p 指向了 main 里的 i
    f(&i);		//先运行函数f  运行后i的值被改为26 接着运行函数g

    // 【传值调用】：此时 i 的值已经被 f 修改成了 26
    // 这里把 26 这个值复制了一份给函数 g 的参数 k
    g(i);

    return 0;
}

// 指针操作函数
void f(int *p) {
    // p 存储的是 i 的地址
    printf(" p=%p\n", p);

    // *p 代表“去 p 指向的地址里看一眼”，此时还没改，所以是 6
    printf("*p=%d\n", *p);

    // 【关键操作】：通过指针直接修改该地址下的数据
    // 这行代码执行后，main 函数里的变量 i 变成了 26
    *p = 26; 
}

// 普通传值函数
void g(int k) {
    // k 只是 i 的一个副本（拷贝）
    // 修改这里的 k 不会影响 main 函数里的 i
    printf("k=%d\n", k);
}
```

**老师的意思是**：`scanf` 是个“没头脑”，它分不清你给它的数字是一个**值**还是一个**坐标**。

**结果**：你给了它值 `6`，它把 `6` 当坐标去寻宝，结果找错了地方，程序通常就直接崩溃（Segment Fault）了。

```c
#include <stdio.h>

// 函数声明：min 和 max 使用指针接收，以便在函数内部修改外部变量的值
void minmax(int a[], int len, int *min, int *max);

int main(void) {
    // 定义一个数组
    int a[] = {1,2,3,4,5,6,7,8,9,12,13,14,16,17,21,23,55};
    int min, max; // 定义用于存放结果的变量

    // 计算数组长度：总字节数 / 单个元素字节数
    int len = sizeof(a) / sizeof(a[0]);

    // 调用函数：注意这里传入的是 &min 和 &max（地址）
    minmax(a, len, &min, &max);

    // 此时 min 和 max 的值已经被函数修改了
    printf("min=%d, max=%d\n", min, max);

    return 0;
}

// 函数定义
void minmax(int a[], int len, int *min, int *max) {
    int i;
    
    // 初始化：假设第一个元素既是最小也是最大
    // *min 代表“去 min 指向的那个内存格子里写数据”
    *min = *max = a[0]; 

    for (i = 1; i < len; i++) {
        // 如果当前元素比记录的最小值还小
        if (a[i] < *min) {
            *min = a[i]; // 更新 main 函数里的 min 变量
        }
        // 如果当前元素比记录的最大值还大
        if (a[i] > *max) {
            *max = a[i]; // 更新 main 函数里的 max 变量
        }
    }
}
```

```c
#include <stdio.h>

/**
 * @param a 被除数
 * @param b 除数
 * @param result 指向保存结果变量的指针
 * @return 如果除法成功（b不为0），返回1；否则返回0
 */
int divide(int a, int b, int *result);

int main(void) {
    int a = 5;
    int b = 2;
    int c; // 准备用来存商的变量

    // 调用函数时，传入 c 的地址 &c
    // 如果 divide 返回 1，if 条件成立
    if ( divide(a, b, &c) ) {
        printf("%d/%d=%d\n", a, b, c);
    } else {
        printf("错误：除数不能为0\n");
    }

    return 0;
}

int divide(int a, int b, int *result) {
    int ret = 1; // 默认成功状态为 1

    if ( b == 0 ) {
        ret = 0; // 如果除数是 0，状态设为失败
    } else {
        // 通过指针解引用，把计算结果直接写回 main 函数的变量 c 中
        *result = a / b;
    }

    return ret; // 返回执行状态
}
```

## 指针与数组

![](https://raw.githubusercontent.com/czh010/images/main/hardwa/image-20260403222749362.png)



```c
#include <stdio.h>

// 这里的 int a[] 看起来是数组，本质上就是 int *a
void minmax(int a[], int len, int *min, int *max);

int main(void) {
    int a[] = {1,2,3,4,5,6,7,8,9,12,13,14,16,17,21,23,55};
    int min, max;

    // 在这里 a 是数组，sizeof(a) 是 68
    printf("main sizeof(a)=%lu\n", sizeof(a));
    printf("main a=%p\n", a);

    // 必须在这里算好长度传过去
    minmax(a, sizeof(a)/sizeof(a[0]), &min, &max);

    printf("min=%d, max=%d\n", min, max);
    return 0;
}

void minmax(int a[], int len, int *min, int *max) {
    // 到了这里，a 已经退化成了指针，sizeof(a) 只是指针的大小（4 或 8）
    printf("minmax sizeof(a)=%lu\n", sizeof(a));
    // 但 a 存储的地址依然指向 main 里的那个数组
    printf("minmax a=%p\n", a);

    *min = *max = a[0]; 
    for (int i = 1; i < len; i++) {
        if (a[i] < *min) *min = a[i];
        if (a[i] > *max) *max = a[i];
    }
}
```

**简单一句话总结：** 数组进函数就变成了“名片”（指针），名片上写着数组的地址，但名片本身可没长得跟整个数组一样大。

**数组在作为函数参数传递时，会“退化”为指针。**

![image-20260403222851707](https://raw.githubusercontent.com/czh010/images/main/hardwa/image-20260403222851707.png)

别担心，这确实是 C 语言中最让新手头疼的“逻辑陷阱”之一。我们换个生活中的场景，不聊代码，聊**“搬家”**，你一听就懂了。

------

### 1. 为什么 `sizeof` 会变？（搬家 vs 寄名片）

想象你在 `main` 函数里有一套**大房子**（数组 `a`），里面有 17 个房间。

- **在 `main` 内部**：

  你就在房子里。如果你问：“我的资产有多大？”（`sizeof(a)`），你会得到整个房子的面积（**68 平方米**）。

- **在调用函数 `minmax` 时**：

  C 语言为了省事，它**不会**把整座房子克隆一份搬给函数（那太累了，浪费内存）。

  它只给函数寄了一张**名片**（指针 `a`），上面写着房子的地址。

- **在 `minmax` 内部**：

  函数手里只拿着这张名片。如果你问函数：“你手里的东西有多大？”（`sizeof(a)`），它量了一下**名片**的大小，告诉你：“只有 **4 厘米**长。”

------

### 2. 既然只有名片，函数怎么改房子里的东西？

虽然函数手里只有名片，但名片上有**地址**啊！

- 函数看到名片上的地址是 `0xbff67d10`。
- 它顺着这个地址，“瞬移”到了你 `main` 里的那座大房子门口。
- 它进屋去检查每个房间（`a[0]`, `a[1]`...），并把找到的最小值和最大值记下来。

**所以：** 地址（`%p`）是一样的，因为它们指向的是**同一座房子**。但 `sizeof` 不一样，因为一个是**量房子**，一个是**量名片**。

------

### 3. 为什么必须传 `len`？（关键所在！）

这就是最有难度的地方：

因为函数手里只有一张名片，它**并不知道**这张名片背后的房子到底有几个房间。

- 如果没有 `len`，函数进了屋，检查完第 17 个房间后，它不知道结束了。
- 它可能会继续往后走，闯进邻居家里（内存越界），导致程序报错或崩溃。
- 所以，你必须在寄名片的时候，顺便打个电话告诉它：“我这房子有 **17 个房间**（`len=17`），你看完这 17 个就回来。”

------

### 4. 一句话总结这个“陷阱”

> **数组只要跨进函数的门槛，就会瞬间“瘦身”成一个指针。**

- **瘦身前**：它知道自己有多长（`sizeof` 是总大小）。
- **瘦身后**：它只剩下一个地址（`sizeof` 是指针大小）。

**你现在回过头看那张截图：**

`main` 里的 `sizeof` 是 68（房子全貌），`minmax` 里的 `sizeof` 是 4（名片大小）。

这个“房子与名片”的比喻，有没有让你感觉稍微透彻一点了？
